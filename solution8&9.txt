
Define a EXPLICIT cursor to display the ssn, fname and dependent names of employees.

DECLARE
c_fname  COMP_MAH4002.Fname%type;
c_SSN COMP_MAH4002.SSN%type;
c_dname DEPEDENT1020.De_name%type;
cursor display is
SELECT Fname, SSN, De_name FROM COMP_MAH4002, DEPEDENT1020 where SSN=ESSN;
BEGIN
OPEN display;
LOOP
FETCH display into c_Fname, c_SSN, c_dname;
EXIT WHEN display%notfound;

dbms_output.put_line('Fname is=' || c_fname);
dbms_output.put_line('SSN  is=' || c_SSN);
dbms_output.put_line('depednt name   is=' ||c_dname);
END LOOP;
CLOSE display;
END;
