2.	Write a stored procedure to display the employee name and salary if name of the working department and ssn of the employee is passed


create or replace procedure  display(mssn in COMP_MAH400.SSN%type, mDname in DEPTMENT_MAH400.Dname%type, mname out COMP_MAH400.Fname%type, msal out COMP_MAH400.salary%type)
 is
   
begin
  select Fname, salary into mname, msal from COMP_MAH400, DEPTMENT_MAH400 where Dname=mDname and SSN=mssn;
 end;
/
 declare 
  mname COMP_MAH400.Fname%type;
  msal COMP_MAH400.salary%type;
  begin
 display(987654321, 'Admin', mname, msal);
dbms_output.put_line('name = ' ||mname);
dbms_output.put_line('salary ' ||msal);
  end;
/
